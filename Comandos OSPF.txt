Comandos OSPF:

ROUTER1:
hostname ROUTER1

interface g0/0
ip address 192.168.10.8 255.255.255.0
no shutdown

interface g0/1
ip address 192.168.20.8 255.255.255.0
no shutdown

exit 

show ip interface brief

router ospf 1
network 192.168.10.0 0.0.0.255 area 0
network 192.168.20.0 0.0.0.255 area 0
