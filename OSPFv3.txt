OSPFv3
hostname ROUTER1

ipv6 unicast-routing

interface g0/0
ipv6 address 2001:ABCD:1435:A1::8/64
no shutdown

interface g0/1
ipv6 address 2002:ABCD:1435:B1::8/64
no shutdown

exit

ipv6 unicast-routing
ipv6 router ospf 1
router-id 1.1.1.1

exit 
show ipv6 ospf

ipv6 unicast-routing
interface g0/0
ipv6 ospf 1 area 0
interface g0/1
ipv6 ospf 1 area 0

exit
show ipv6 ospf neighbor
show ipv6 protocols
