hostname ROUTER1

ipv6 unicast-routing

interface g0/0
ipv6 address 2001:ABCD:1435:A1::8/64
no shutdown

interface g0/1
ipv6 address 2002:ABCD:1435:B1::8/64
no shutdown

exit

ipv6 unicast-routing
ipv6 router rip infracom
interface g0/0
ipv6 rip infracom enable
interface g0/1
ipv6 rip infracom enable


exit
show ipv6 protocols
show run
show ipv6 rip database
show ipv6 rip infracom
